- query-table:: true
  #+BEGIN_QUERY
  {:title [:h1 "Tareas repetitivas vencidas" ]
  :query [
    :find (pull ?b [*])
    :in $ ?start ?next
    :where
      [?b :block/scheduled ?d]
      [?p :block/name "tarea repetitiva"]
      [?b :block/refs ?p]
      [(> ?d ?start)]
      [(< ?d ?next)]
  ]
  :inputs [:900d-before :today]
  :collapsed? false
  :breadcrumb-show? true}
  #+END_QUERY
